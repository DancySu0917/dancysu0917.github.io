# 用 JS 实现随机选取 10–100 之间的 10 个数字，存入一个数组并排序？（必会）

**题目**: 用 JS 实现随机选取 10–100 之间的 10 个数字，存入一个数组并排序？（必会）

**答案**:

用JavaScript实现随机选取10-100之间的10个数字，存入数组并排序的方法：

```javascript
// 方法1：基础实现
function getRandomNumbers() {
  const numbers = [];
  
  // 生成10个随机数
  for (let i = 0; i < 10; i++) {
    // Math.random() 生成 [0, 1) 的随机数
    // 乘以范围 (100 - 10 + 1) = 91，得到 [0, 91)
    // 加上最小值 10，得到 [10, 101)
    // Math.floor() 向下取整，得到 [10, 100] 的整数
    const randomNum = Math.floor(Math.random() * (100 - 10 + 1)) + 10;
    numbers.push(randomNum);
  }
  
  // 排序（默认升序）
  numbers.sort((a, b) => a - b);
  
  return numbers;
}

console.log(getRandomNumbers()); // 示例输出：[15, 23, 34, 45, 56, 67, 78, 82, 89, 95]

// 方法2：使用Array.from简化
function getRandomNumbers2() {
  const numbers = Array.from({ length: 10 }, () => 
    Math.floor(Math.random() * (100 - 10 + 1)) + 10
  );
  
  return numbers.sort((a, b) => a - b);
}

console.log(getRandomNumbers2());

// 方法3：去重版本（确保10个数字都不相同）
function getUniqueRandomNumbers() {
  const numbers = new Set();
  
  // 持续生成随机数直到集合中有10个不重复的数字
  while (numbers.size < 10) {
    const randomNum = Math.floor(Math.random() * (100 - 10 + 1)) + 10;
    numbers.add(randomNum);
  }
  
  // 转换为数组并排序
  return Array.from(numbers).sort((a, b) => a - b);
}

console.log(getUniqueRandomNumbers());

// 方法4：Fisher-Yates 洗牌算法（从所有可能数字中随机选择10个）
function getRandomNumbersByShuffle() {
  // 创建包含10-100所有数字的数组
  const allNumbers = [];
  for (let i = 10; i <= 100; i++) {
    allNumbers.push(i);
  }
  
  // Fisher-Yates 洗牌算法的前10个元素
  for (let i = allNumbers.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [allNumbers[i], allNumbers[j]] = [allNumbers[j], allNumbers[i]];
  }
  
  // 取前10个数字并排序
  return allNumbers.slice(0, 10).sort((a, b) => a - b);
}

console.log(getRandomNumbersByShuffle());

// 方法5：更通用的实现，支持自定义参数
function getRandomNumbersCustom(min = 10, max = 100, count = 10, unique = false) {
  const numbers = unique ? new Set() : [];
  
  if (unique && (max - min + 1) < count) {
    throw new Error(`无法在范围[${min}, ${max}]内生成${count}个不重复的数字`);
  }
  
  while (unique ? numbers.size < count : numbers.length < count) {
    const randomNum = Math.floor(Math.random() * (max - min + 1)) + min;
    
    if (unique) {
      numbers.add(randomNum);
    } else {
      numbers.push(randomNum);
    }
  }
  
  const result = unique ? Array.from(numbers) : numbers;
  return result.sort((a, b) => a - b);
}

// 使用示例
console.log(getRandomNumbersCustom(10, 100, 10, false)); // 允许重复
console.log(getRandomNumbersCustom(10, 100, 10, true));  // 不允许重复

// 方法6：函数式编程风格
const getRandomNumbersFunctional = (min = 10, max = 100, count = 10) => {
  const generateRandom = () => Math.floor(Math.random() * (max - min + 1)) + min;
  
  return Array.from({ length: count }, generateRandom)
    .sort((a, b) => a - b);

console.log(getRandomNumbersFunctional());

// 随机数生成函数封装
function getRandomInt(min, max) {
  min = Math.ceil(min);
  max = Math.floor(max);
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

// 最终推荐实现
function generateRandomNumbers() {
  const numbers = [];
  
  for (let i = 0; i < 10; i++) {
    numbers.push(getRandomInt(10, 100));
  }
  
  // 升序排序
  return numbers.sort((a, b) => a - b);
  
  // 如果需要降序排序，使用：return numbers.sort((a, b) => b - a);
}

// 测试函数
function testRandomness() {
  for (let i = 0; i < 3; i++) {
    console.log(`第${i+1}次生成:`, generateRandomNumbers());
  }
}

testRandomness();
```

**关键点说明**：
1. **随机数生成**：`Math.random()` 生成 [0, 1) 的浮点数，需通过计算转换为指定范围的整数
2. **范围计算**：`Math.floor(Math.random() * (max - min + 1)) + min` 确保包含边界值
3. **排序方法**：使用 `Array.prototype.sort()` 配合比较函数 `(a, b) => a - b` 进行升序排序
4. **去重处理**：使用 `Set` 数据结构确保生成不重复的数字
5. **算法选择**：Fisher-Yates洗牌算法适用于从大量数据中随机选择部分元素

**扩展应用**：
- 可以根据需要调整随机数范围和数量
- 可以选择是否允许重复数字
- 可以选择升序或降序排列
