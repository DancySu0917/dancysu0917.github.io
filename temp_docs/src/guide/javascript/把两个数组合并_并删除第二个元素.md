# 把两个数组合并，并删除第二个元素？（必会）

**题目**: 把两个数组合并，并删除第二个元素？（必会）

**答案**:

要将两个数组合并并删除合并后数组的第二个元素，有多种实现方法：

## 1. 使用扩展运算符和数组方法

### 方法一：使用 slice() 方法
```javascript
function mergeAndRemoveSecond(arr1, arr2) {
  // 合并数组
  const merged = [...arr1, ...arr2];
  
  // 删除第二个元素（索引为1）
  return merged.slice(0, 1).concat(merged.slice(2));
}

// 示例
const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
const result = mergeAndRemoveSecond(array1, array2);
console.log(result); // [1, 3, 4, 5, 6]
```

### 方法二：使用 filter() 方法
```javascript
function mergeAndRemoveSecond(arr1, arr2) {
  const merged = [...arr1, ...arr2];
  
  // 过滤掉索引为1的元素（第二个元素）
  return merged.filter((item, index) => index !== 1);
}

// 示例
const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
const result = mergeAndRemoveSecond(array1, array2);
console.log(result); // [1, 3, 4, 5, 6]
```

## 2. 使用 splice() 方法

```javascript
function mergeAndRemoveSecond(arr1, arr2) {
  // 合并数组
  const merged = [...arr1, ...arr2];
  
  // 删除索引为1的元素（第二个元素）
  merged.splice(1, 1);
  
  return merged;
}

// 示例
const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
const result = mergeAndRemoveSecond(array1, array2);
console.log(result); // [1, 3, 4, 5, 6]
```

## 3. 使用解构赋值

```javascript
function mergeAndRemoveSecond(arr1, arr2) {
  const merged = [...arr1, ...arr2];
  
  // 使用解构赋值跳过第二个元素
  const [first, , ...rest] = merged;
  return [first, ...rest];
}

// 示例
const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
const result = mergeAndRemoveSecond(array1, array2);
console.log(result); // [1, 3, 4, 5, 6]
```

## 4. 使用 Array.concat() 方法

```javascript
function mergeAndRemoveSecond(arr1, arr2) {
  // 使用 concat 合并数组
  const merged = arr1.concat(arr2);
  
  // 删除第二个元素
  return merged.filter((item, index) => index !== 1);
}

// 示例
const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
const result = mergeAndRemoveSecond(array1, array2);
console.log(result); // [1, 3, 4, 5, 6]
```

## 5. 一次性操作（更高效）

```javascript
function mergeAndRemoveSecond(arr1, arr2) {
  // 如果 arr1 长度大于等于2，删除 arr1 的第二个元素
  if (arr1.length >= 2) {
    return [...arr1.slice(0, 1), ...arr1.slice(2), ...arr2];
  } else if (arr1.length === 1) {
    // 如果 arr1 只有一个元素，删除 arr2 的第一个元素（即合并后数组的第二个元素）
    return [arr1[0], ...arr2.slice(1)];
  } else {
    // 如果 arr1 为空，删除 arr2 的第二个元素
    return [arr2[0], ...arr2.slice(2)];
  }
}

// 示例
const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
const result = mergeAndRemoveSecond(array1, array2);
console.log(result); // [1, 3, 4, 5, 6]
```

## 6. 通用解决方案（处理边界情况）

```javascript
function mergeAndRemoveSecond(arr1, arr2) {
  // 合并数组
  const merged = [...arr1, ...arr2];
  
  // 检查数组长度，如果长度小于2，返回原数组
  if (merged.length < 2) {
    return merged;
  }
  
  // 删除第二个元素（索引为1）
  return merged.filter((item, index) => index !== 1);
}

// 测试边界情况
console.log(mergeAndRemoveSecond([1], [2, 3])); // [1, 3] - 合并后删除第二个元素
console.log(mergeAndRemoveSecond([1], []));      // [1] - 只有一个元素
console.log(mergeAndRemoveSecond([], [1, 2]));  // [2] - 删除第二个元素后只剩一个
console.log(mergeAndRemoveSecond([], []));       // [] - 空数组
```

## 性能对比

- **splice() 方法**：直接修改原数组，内存效率高，但会改变原数组
- **filter() 方法**：创建新数组，不会修改原数组，代码清晰
- **slice() 方法**：创建新数组，适合需要保留原数组不变的场景
- **解构赋值**：代码简洁，但只适合删除固定位置的元素

根据具体需求选择合适的方法，一般推荐使用 `filter()` 方法，因为它不会修改原数组，代码也相对清晰。
